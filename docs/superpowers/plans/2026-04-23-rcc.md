# rcc Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a PyPI-distributable Python CLI (`rcc`) that generalizes two existing bash scripts (`run.sh`, `sync.sh`) into a reusable tool with profile-based config, `push`/`pull`/`run`/`shell`/`status`/`close`/`config`/`init` subcommands, primary OpenSSH+rsync transport, paramiko fallback.

**Architecture:** Single-package `src/rcc/` layout. Thin `commands/` wrappers delegate to pure-logic modules: `config.py` (discovery + profile merge), `ssh.py` (arg builder, mux, exec), `sync.py` (rsync + paramiko SFTP), `ignore.py` (gitignore-syntax). Typer `cli.py` is the entry point and the only layer that converts typed exceptions into exit codes.

**Tech Stack:** Python 3.11+, `uv` for packaging, `typer` for CLI, `paramiko` for SSH fallback, `pytest` + `pytest-cov` + `ruff`.

**Spec:** `docs/superpowers/specs/2026-04-23-rcc-design.md`

**Why this decomposition:**
- `_rsync.py` and `_paramiko_fallback.py` are private siblings of `sync.py` so the two transport implementations can be unit-tested independently without widening `sync.py`'s surface.
- `commands/*.py` stay thin by construction: they only do flag parsing + profile resolution + delegation.
- `ssh.py` bundles arg-builder + exec because splitting would force most tests to import both.

```python
class RccError(Exception):

class MissingDependencyError(RccError):
    """A required transport (ssh, rsync, paramiko) is unavailable."""

        super().__init__(message)
```

```python
from __future__ import annotations

import fnmatch
from dataclasses import dataclass
from pathlib import Path, PurePosixPath


@dataclass(frozen=True)
class _Rule:
    pattern: str
    negate: bool
    dir_only: bool
    root_anchored: bool


class GitignoreMatcher:
    """Subset of gitignore syntax: trailing-slash dirs, *-globs, leading-slash anchor, ! negation."""

    def __init__(self, patterns: list[str]) -> None:
        self._rules: list[_Rule] = [r for r in map(self._parse, patterns) if r is not None]

    @classmethod
    def from_file(cls, path: Path) -> "GitignoreMatcher":
        return cls(path.read_text().splitlines())

    @staticmethod
    def _parse(raw: str) -> _Rule | None:
        line = raw.strip()
        if not line or line.startswith("#"):
            return None
        negate = line.startswith("!")
        if negate:
            line = line[1:]
        root_anchored = line.startswith("/")
        if root_anchored:
            line = line[1:]
        dir_only = line.endswith("/")
        if dir_only:
            line = line[:-1]
        return _Rule(pattern=line, negate=negate, dir_only=dir_only, root_anchored=root_anchored)

    def match(self, path: Path | str, *, is_dir: bool) -> bool:
        p = PurePosixPath(str(path).replace("\\", "/"))
        parts = p.parts
        matched = False
        for rule in self._rules:
            if rule.dir_only and not is_dir and not self._has_dir_ancestor_matching(parts, rule):
                if not self._path_matches(parts, rule):
                    continue
            if self._path_matches(parts, rule):
                matched = not rule.negate
        return matched

    @staticmethod
    def _path_matches(parts: tuple[str, ...], rule: _Rule) -> bool:
        if rule.root_anchored:
            candidates = ["/".join(parts[:i]) for i in range(1, len(parts) + 1)]
            return any(fnmatch.fnmatchcase(c, rule.pattern) for c in candidates[:1])
        return any(fnmatch.fnmatchcase(part, rule.pattern) for part in parts) or fnmatch.fnmatchcase(
            "/".join(parts), rule.pattern
        )

    @staticmethod
    def _has_dir_ancestor_matching(parts: tuple[str, ...], rule: _Rule) -> bool:
        for i in range(len(parts)):
            if rule.root_anchored and i > 0:
                break
            if fnmatch.fnmatchcase(parts[i], rule.pattern):
                return True
        return False
```

```python
from __future__ import annotations

import tomllib
from dataclasses import dataclass, field
from pathlib import Path

from rcc.errors import ConfigError


@dataclass(frozen=True)
class Profile:
    host: str
    remote_dir: str
    ssh_control_persist: str = "30m"
    ssh_control_dir: str = "~/.ssh/controlmasters"


@dataclass(frozen=True)
class Config:
    default: str
    profiles: dict[str, Profile] = field(default_factory=dict)


_REQUIRED = ("host", "remote_dir")


def load_config(path: Path) -> Config:
    try:
        raw = tomllib.loads(path.read_text())
    except tomllib.TOMLDecodeError as e:
        raise ConfigError(f"{path}: malformed TOML: {e}") from e
    except FileNotFoundError as e:
        raise ConfigError(f"{path}: not found") from e

    default = raw.get("default")
    if not isinstance(default, str):
        raise ConfigError(f"{path}: missing 'default' profile name")

    profiles_raw = raw.get("profiles", {})
    if not profiles_raw:
        raise ConfigError(f"{path}: no profiles defined")

    profiles: dict[str, Profile] = {}
    for name, body in profiles_raw.items():
        for req in _REQUIRED:
            if req not in body:
                raise ConfigError(f"{path}: profile '{name}' missing required field '{req}'")
        profiles[name] = Profile(
            host=body["host"],
            remote_dir=body["remote_dir"],
            ssh_control_persist=body.get("ssh_control_persist", "30m"),
            ssh_control_dir=body.get("ssh_control_dir", "~/.ssh/controlmasters"),
        )

    if default not in profiles:
        raise ConfigError(f"{path}: default profile '{default}' not defined")

    return Config(default=default, profiles=profiles)
```

```python
def find_rcc_dir(start: Path) -> Path | None:
    """Walk up from `start` until a .rcc/ directory is found. Returns None if not found."""
    current = start.resolve()
    while True:
        candidate = current / ".rcc"
        if candidate.is_dir():
            return candidate
        if current.parent == current:
            return None
        current = current.parent
```

```python
import dataclasses


def resolve_profile(
    start: Path,
    *,
    profile_name: str | None = None,
    host_override: str | None = None,
    remote_dir_override: str | None = None,
) -> Profile:
    rcc_dir = find_rcc_dir(start)
    if rcc_dir is None:
        raise ConfigError("no .rcc/ found (run 'rcc init')")
    cfg = load_config(rcc_dir / "config.toml")
    name = profile_name or cfg.default
    if name not in cfg.profiles:
        raise ConfigError(f"profile '{name}' not defined in {rcc_dir}/config.toml")
    profile = cfg.profiles[name]
    overrides: dict[str, str] = {}
    if host_override:
        overrides["host"] = host_override
    if remote_dir_override:
        overrides["remote_dir"] = remote_dir_override
    if overrides:
        profile = dataclasses.replace(profile, **overrides)
    return profile
```

```python
from __future__ import annotations

import shlex
from typing import Literal

from rcc.config import Profile

Mode = Literal["run", "mux_check", "mux_exit"]


def _shared_mux_options(profile: Profile) -> list[str]:
    return [
        "-o", f"ControlPersist={profile.ssh_control_persist}",
        "-o", f"ControlPath={profile.ssh_control_dir}/%C",
        "-o", "ControlMaster=auto",
        "-o", "LogLevel=ERROR",
    ]


def build_ssh_args(profile: Profile, *, mode: Mode, tty: bool = False) -> list[str]:
    argv: list[str] = ["ssh", *_shared_mux_options(profile)]
    if mode == "run":
        if tty:
            argv.append("-t")
        argv.append(profile.host)
        return argv
    if mode == "mux_check":
        argv.extend(["-O", "check", profile.host])
        return argv
    if mode == "mux_exit":
        argv.extend(["-O", "exit", profile.host])
        return argv
    raise ValueError(f"unknown mode: {mode}")


def build_rsync_e_string(profile: Profile) -> str:
    return "ssh " + " ".join(shlex.quote(p) for p in _shared_mux_options(profile))


def build_remote_command(remote_dir: str, argv: list[str]) -> str:
    quoted_dir = shlex.quote(remote_dir)
    missing_msg = shlex.quote(f"error: remote directory does not exist: {remote_dir}")
    cmd_str = " ".join(shlex.quote(a) for a in argv) if argv else "exec bash -l"
    return (
        "set -euo pipefail; "
        f"if [[ ! -d {quoted_dir} ]]; then printf '%s\\n' {missing_msg} >&2; exit 1; fi; "
        f"cd -- {quoted_dir}; "
        f"{cmd_str}"
    )
```

```python
import logging
import os
import shlex as _shlex
import shutil
import subprocess
from pathlib import Path

from rcc.errors import MissingDependencyError, RemoteError

log = logging.getLogger(__name__)


def _ensure_control_dir(profile: Profile) -> None:
    p = Path(os.path.expanduser(profile.ssh_control_dir))
    p.mkdir(parents=True, exist_ok=True)
    try:
        p.chmod(0o700)
    except OSError as e:
        log.warning("failed to chmod %s to 0700: %s", p, e)


def _primary_available() -> bool:
    return shutil.which("ssh") is not None


def run_remote(profile: Profile, argv: list[str], *, tty: bool = False) -> int:
    """Run argv on remote inside profile.remote_dir. Returns remote exit code."""
    if _primary_available():
        _ensure_control_dir(profile)
        ssh_argv = build_ssh_args(profile, mode="run", tty=tty)
        remote_cmd = build_remote_command(profile.remote_dir, argv)
        full = ssh_argv + ["bash", "-lc", remote_cmd]
        return subprocess.run(full).returncode
    log.warning("ssh not on PATH; using paramiko fallback")
    return _paramiko_run_remote(profile, argv, tty=tty)


def ensure_remote_dir(profile: Profile) -> None:
    """mkdir -p remote_dir on the host."""
    if _primary_available():
        _ensure_control_dir(profile)
        ssh_argv = build_ssh_args(profile, mode="run", tty=False)
        cmd = f"mkdir -p -- {_shlex.quote(profile.remote_dir)}"
        full = ssh_argv + ["bash", "-lc", cmd]
        result = subprocess.run(full)
        if result.returncode != 0:
            raise RemoteError(
                f"failed to create remote dir {profile.remote_dir}", exit_code=result.returncode
            )
        return
    log.warning("ssh not on PATH; using paramiko fallback")
    _paramiko_ensure_remote_dir(profile)


def mux_check(profile: Profile) -> bool:
    if not _primary_available():
        return False
    _ensure_control_dir(profile)
    result = subprocess.run(
        build_ssh_args(profile, mode="mux_check"),
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
    )
    return result.returncode == 0


def mux_exit(profile: Profile) -> None:
    if not _primary_available():
        raise MissingDependencyError("ControlMaster not available without system ssh")
    _ensure_control_dir(profile)
    subprocess.run(build_ssh_args(profile, mode="mux_exit"), stdout=subprocess.DEVNULL)


def _paramiko_run_remote(profile: Profile, argv: list[str], *, tty: bool) -> int:
    try:
        import paramiko
    except ImportError as e:
        raise MissingDependencyError("paramiko required for fallback; install paramiko") from e
    client = paramiko.SSHClient()
    client.load_system_host_keys()
    client.connect(hostname=profile.host)
    try:
        remote_cmd = build_remote_command(profile.remote_dir, argv)
        full_cmd = f"bash -lc {_shlex.quote(remote_cmd)}"
        transport = client.get_transport()
        if transport is None:
            raise RemoteError("paramiko: could not open transport")
        channel = transport.open_session()
        if tty:
            channel.get_pty()
        channel.exec_command(full_cmd)
        while True:
            if channel.recv_ready():
                data = channel.recv(4096)
                if not data:
                    break
                import sys
                sys.stdout.buffer.write(data)
                sys.stdout.buffer.flush()
            if channel.recv_stderr_ready():
                err = channel.recv_stderr(4096)
                import sys
                sys.stderr.buffer.write(err)
                sys.stderr.buffer.flush()
            if channel.exit_status_ready() and not channel.recv_ready() and not channel.recv_stderr_ready():
                break
        return channel.recv_exit_status()
    finally:
        client.close()


def _paramiko_ensure_remote_dir(profile: Profile) -> None:
    try:
        import paramiko
    except ImportError as e:
        raise MissingDependencyError("paramiko required for fallback; install paramiko") from e
    client = paramiko.SSHClient()
    client.load_system_host_keys()
    client.connect(hostname=profile.host)
    try:
        stdin, stdout, stderr = client.exec_command(
            f"mkdir -p -- {_shlex.quote(profile.remote_dir)}"
        )
        code = stdout.channel.recv_exit_status()
        if code != 0:
            raise RemoteError(
                f"failed to create remote dir {profile.remote_dir}", exit_code=code
            )
    finally:
        client.close()
```

```python
from __future__ import annotations

import subprocess
from pathlib import Path

from rcc.errors import RemoteError


def build_rsync_argv(
    *,
    source: Path | str,
    destination: str,
    e_string: str,
    exclude_from: Path,
    extra_excludes: list[str],
    dry_run: bool,
    delete: bool,
) -> list[str]:
    argv: list[str] = [
        "rsync",
        "-a",
        "-z",
        "--human-readable",
        "--info=stats2,progress2",
        "--partial",
        f"--exclude-from={exclude_from}",
    ]
    for pat in extra_excludes:
        argv.append(f"--exclude={pat}")
    if dry_run:
        argv.extend(["--dry-run", "--itemize-changes"])
    if delete:
        argv.append("--delete")
    argv.extend(["-e", e_string])
    src_str = f"{source}/" if not str(source).endswith("/") else str(source)
    dst_str = destination if destination.endswith("/") else f"{destination}/"
    argv.extend([src_str, dst_str])
    return argv


def run_rsync(argv: list[str]) -> None:
    result = subprocess.run(argv)
    if result.returncode != 0:
        raise RemoteError(f"rsync exited with code {result.returncode}", exit_code=result.returncode)
```

```python
from __future__ import annotations

from dataclasses import dataclass
from pathlib import Path

from rcc.ignore import GitignoreMatcher


@dataclass(frozen=True)
class TransferPlan:
    absolute_path: Path
    relative_path: str


def _build_matcher(exclude_from: Path, extra_excludes: list[str]) -> GitignoreMatcher:
    lines: list[str] = []
    if exclude_from.exists():
        lines.extend(exclude_from.read_text().splitlines())
    lines.extend(extra_excludes)
    return GitignoreMatcher(lines)


def plan_push_transfers(
    source: Path, *, exclude_from: Path, extra_excludes: list[str]
) -> list[TransferPlan]:
    matcher = _build_matcher(exclude_from, extra_excludes)
    plans: list[TransferPlan] = []
    for path in source.rglob("*"):
        if not path.is_file():
            continue
        rel = path.relative_to(source).as_posix()
        if matcher.match(rel, is_dir=False):
            continue
        plans.append(TransferPlan(absolute_path=path, relative_path=rel))
    return plans


def plan_pull_transfers(
    sftp, remote_root: str, *, exclude_from: Path, extra_excludes: list[str]
) -> list[TransferPlan]:
    """Walk remote via SFTP and return transfer plans (caller handles .get)."""
    import stat

    matcher = _build_matcher(exclude_from, extra_excludes)
    plans: list[TransferPlan] = []

    def walk(prefix: str) -> None:
        for entry in sftp.listdir_attr(f"{remote_root}/{prefix}" if prefix else remote_root):
            rel = f"{prefix}/{entry.filename}" if prefix else entry.filename
            is_dir = stat.S_ISDIR(entry.st_mode)
            if matcher.match(rel, is_dir=is_dir):
                continue
            if is_dir:
                walk(rel)
            else:
                plans.append(TransferPlan(absolute_path=Path(f"{remote_root}/{rel}"), relative_path=rel))

    walk("")
    return plans
```

```python
from __future__ import annotations

import logging
import shutil
from pathlib import Path

from rcc.config import Profile
from rcc.errors import MissingDependencyError
from rcc._paramiko_fallback import plan_pull_transfers, plan_push_transfers
from rcc._rsync import build_rsync_argv, run_rsync
from rcc.ssh import build_rsync_e_string, ensure_remote_dir

log = logging.getLogger(__name__)


def _rccignore_path(project_dir: Path) -> Path:
    return project_dir / ".rcc" / "rccignore"


def _primary_available() -> bool:
    return shutil.which("ssh") is not None and shutil.which("rsync") is not None


def push(
    *,
    project_dir: Path,
    profile: Profile,
    dry_run: bool,
    delete: bool,
    extra_excludes: list[str],
) -> None:
    if _primary_available():
        ensure_remote_dir(profile)
        argv = build_rsync_argv(
            source=project_dir,
            destination=f"{profile.host}:{profile.remote_dir}",
            e_string=build_rsync_e_string(profile),
            exclude_from=_rccignore_path(project_dir),
            extra_excludes=extra_excludes,
            dry_run=dry_run,
            delete=delete,
        )
        run_rsync(argv)
        return
    log.warning("rsync/ssh not on PATH; using paramiko fallback")
    _paramiko_push(project_dir, profile, dry_run=dry_run, delete=delete, extra_excludes=extra_excludes)


def pull(
    *,
    project_dir: Path,
    profile: Profile,
    dry_run: bool,
    delete: bool,
    extra_excludes: list[str],
) -> None:
    if _primary_available():
        argv = build_rsync_argv(
            source=f"{profile.host}:{profile.remote_dir}",
            destination=str(project_dir),
            e_string=build_rsync_e_string(profile),
            exclude_from=_rccignore_path(project_dir),
            extra_excludes=extra_excludes,
            dry_run=dry_run,
            delete=delete,
        )
        run_rsync(argv)
        return
    log.warning("rsync/ssh not on PATH; using paramiko fallback")
    _paramiko_pull(project_dir, profile, dry_run=dry_run, delete=delete, extra_excludes=extra_excludes)


def _paramiko_push(
    project_dir: Path, profile: Profile, *, dry_run: bool, delete: bool, extra_excludes: list[str]
) -> None:
    try:
        import paramiko
    except ImportError as e:
        raise MissingDependencyError("paramiko required for fallback sync; install paramiko") from e
    plans = plan_push_transfers(
        project_dir,
        exclude_from=_rccignore_path(project_dir),
        extra_excludes=extra_excludes,
    )
    if dry_run:
        for p in plans:
            log.info("would transfer %s", p.relative_path)
        return
    client = paramiko.SSHClient()
    client.load_system_host_keys()
    client.connect(hostname=profile.host)
    try:
        sftp = client.open_sftp()
        try:
            _sftp_mkdir_p(sftp, profile.remote_dir)
            for p in plans:
                remote_path = f"{profile.remote_dir}/{p.relative_path}"
                _sftp_mkdir_p(sftp, remote_path.rsplit("/", 1)[0])
                sftp.put(str(p.absolute_path), remote_path)
            if delete:
                _sftp_prune(sftp, profile.remote_dir, {p.relative_path for p in plans})
        finally:
            sftp.close()
    finally:
        client.close()


def _paramiko_pull(
    project_dir: Path, profile: Profile, *, dry_run: bool, delete: bool, extra_excludes: list[str]
) -> None:
    try:
        import paramiko
    except ImportError as e:
        raise MissingDependencyError("paramiko required for fallback sync; install paramiko") from e
    client = paramiko.SSHClient()
    client.load_system_host_keys()
    client.connect(hostname=profile.host)
    try:
        sftp = client.open_sftp()
        try:
            plans = plan_pull_transfers(
                sftp,
                profile.remote_dir,
                exclude_from=_rccignore_path(project_dir),
                extra_excludes=extra_excludes,
            )
            if dry_run:
                for p in plans:
                    log.info("would download %s", p.relative_path)
                return
            for p in plans:
                local_target = project_dir / p.relative_path
                local_target.parent.mkdir(parents=True, exist_ok=True)
                sftp.get(f"{profile.remote_dir}/{p.relative_path}", str(local_target))
            if delete:
                existing = {p.relative_path for p in plans}
                for path in project_dir.rglob("*"):
                    if not path.is_file():
                        continue
                    rel = path.relative_to(project_dir).as_posix()
                    if rel.startswith(".rcc/"):
                        continue
                    if rel not in existing:
                        path.unlink()
        finally:
            sftp.close()
    finally:
        client.close()


def _sftp_mkdir_p(sftp, path: str) -> None:
    parts = [p for p in path.split("/") if p]
    cur = "" if path.startswith("/") else "."
    for part in parts:
        cur = f"{cur}/{part}" if cur and cur != "." else ("/" + part if path.startswith("/") else part)
        try:
            sftp.stat(cur)
        except IOError:
            sftp.mkdir(cur)


def _sftp_prune(sftp, remote_root: str, keep: set[str]) -> None:
    import stat

    def walk(prefix: str) -> None:
        listing = sftp.listdir_attr(f"{remote_root}/{prefix}" if prefix else remote_root)
        for entry in listing:
            rel = f"{prefix}/{entry.filename}" if prefix else entry.filename
            full = f"{remote_root}/{rel}"
            if stat.S_ISDIR(entry.st_mode):
                walk(rel)
                try:
                    sftp.rmdir(full)
                except IOError:
                    pass
            elif rel not in keep:
                sftp.remove(full)

    walk("")
```

```python
    verbose: bool = typer.Option(False, "-v", "--verbose", help="Verbose logging"),

    context_settings={"allow_extra_args": True, "ignore_unknown_options": True},

app.command(name="shell")(shell_cmd.shell)
app.command(name="status")(status_cmd.status)
app.command(name="close")(close_cmd.close)
app.command(name="config")(config_cmd.config)

    typer.echo(__version__)

    except ConfigError as e:

        typer.echo(f"error: {e}", err=True)
        sys.exit(1)

        sys.exit(1)
```

```python
from __future__ import annotations

import os
import shutil
import subprocess
from pathlib import Path

import pytest

pytestmark = pytest.mark.integration


def _localhost_ssh_works() -> bool:
    if shutil.which("ssh") is None:
        return False
    r = subprocess.run(
        ["ssh", "-o", "BatchMode=yes", "-o", "ConnectTimeout=3", "localhost", "true"],
        capture_output=True,
    )
    return r.returncode == 0


@pytest.mark.skipif(not _localhost_ssh_works(), reason="ssh localhost not available")
def test_push_run_pull_roundtrip(tmp_path: Path):
    local = tmp_path / "local"
    remote = tmp_path / "remote"
    local.mkdir()
    (local / "hello.txt").write_text("hi")
    rcc_dir = local / ".rcc"
    rcc_dir.mkdir()
    (rcc_dir / "config.toml").write_text(
        f'default = "lh"\n'
        f'[profiles.lh]\n'
        f'host = "localhost"\n'
        f'remote_dir = "{remote}"\n'
    )
    (rcc_dir / "rccignore").write_text("")
    env = {**os.environ}
    r = subprocess.run(["uv", "run", "rcc", "push"], cwd=local, env=env, capture_output=True, text=True)
    assert r.returncode == 0, r.stderr
    assert (remote / "hello.txt").read_text() == "hi"

    r = subprocess.run(
        ["uv", "run", "rcc", "run", "--", "cat", "hello.txt"],
        cwd=local,
        env=env,
        capture_output=True,
        text=True,
    )
    assert r.returncode == 0
    assert "hi" in r.stdout
```
